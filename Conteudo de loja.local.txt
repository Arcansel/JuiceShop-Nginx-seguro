server {
    listen 80;
    server_name loja.local;

    return 301 https://$host:8443$request_uri;
}

server {
    listen 8443 ssl;
    server_name loja.local;

    ssl_certificate /etc/nginx/ssl/loja.crt;
    ssl_certificate_key /etc/nginx/ssl/loja.key;

    limit_req_status 429;

    if ($request_method !~ ^(GET|POST|HEAD)$ ) {
        return 405;
    }

    access_log /var/log/nginx/loja_access.log;
    error_log /var/log/nginx/loja_error.log;

    if ($host !~ ^loja\.local$) {
        return 444;
    }

    location ~ /\.(env|git|htaccess|htpasswd|log|bak|swp)$ {
        deny all;
        return 403;
    }

    location ~* (\.\./|%2e%2e%2f|%2e%2e/|..%2f) {
        deny all;
        return 403;
    }

    if ($request_uri ~* "(\.\./|\.\.\\|%2e%2e)") {
        return 403;
    }

    location ~* "%2e%2e" {
        return 403;
    }

    location ^~ /rest/user/login {
        limit_req zone=iplimit burst=5 nodelay;

        proxy_pass http://127.0.0.1:5001;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }

    location / {
        limit_req zone=iplimit burst=20 nodelay;

        proxy_pass http://127.0.0.1:3000;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;

        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";

        proxy_hide_header ETag;

        add_header X-Frame-Options "SAMEORIGIN" always;
        add_header X-Content-Type-Options "nosniff" always;
        add_header Referrer-Policy "strict-origin-when-cross-origin" always;

        add_header X-XSS-Protection "1; mode=block" always;
        add_header Content-Security-Policy "
        default-src 'self';
        script-src 'self' 'unsafe-inline' http://cdnjs.cloudflare.com https://cdnjs.cloudflare.com;
        style-src 'self' 'unsafe-inline' http://cdnjs.cloudflare.com https://cdnjs.cloudflare.com;
        img-src 'self' data:;
        font-src 'self' http://cdnjs.cloudflare.com https://cdnjs.cloudflare.com;
        connect-src 'self' ws:;
        frame-ancestors 'self';
        " always;

	add_header Strict-Transport-Security "max-age=31536000" always;
    }
}
